93impl fmt::Display for ShotLimiting {

95        match self {
96            ShotLimiting::Normal => write!(f, "Normal"),
97            ShotLimiting::Imminent(seconds) => write!(f, "Limiting in {} seconds", seconds),
98            ShotLimiting::Limiting(range) => {
99                write!(f, "Limiting to approximately {}% range", range)
100            }
101            ShotLimiting::Invalid(val) => write!(f, "Invalid shot limiting value: {}", val),
102        }
103    }

125#[derive(Clone, Debug, Eq, PartialEq)]
126pub enum AlertStatus {
127    Normal,
128    Active(u8),
129    Overflow(u8),
130}
131
132impl AlertStatus {
133    pub fn from_flags(flags: u8) -> AlertStatus {
134        if flags & (1 << 6) != 0 {
135            AlertStatus::Overflow(flags & 0b00111111)
136        } else if flags & (1 << 7) != 0 {
137            AlertStatus::Active(flags & 0b00111111)
138        } else {
139            AlertStatus::Normal
140        }
141    }
142}
143
144impl fmt::Display for AlertStatus {
145    fn fmt(&self, f: &mut fmt::Formatter) -> std::fmt::Result {
146        match self {
147            AlertStatus::Normal => write!(f, "Normal"),
148            AlertStatus::Active(cursor) => write!(f, "Alert: {}", cursor),
149            AlertStatus::Overflow(cursor) => write!(f, "Alert (overflow): {}", cursor),
150        }
151    }
152}

154#[derive(Clone, Debug, Eq, PartialEq)]
155pub struct Header {
156    /// Identifies lidar data vs. other packets in stream.
157    /// Packet Type is 0x1 for Lidar packets.
158    pub packet_type: u16,
159    /// Index of the lidar scan, increments every time the sensor completes a
160    /// rotation, crossing the zero azimuth angle.
161    pub frame_id: u16,
162    /// Initialization ID. Updates on every reinit, which may be triggered by
163    /// the user or an error, and every reboot.
164    pub init_id: u32,
165    /// Serial number of the sensor. This value is unique to each sensor and
166    /// can be found on a sticker affixed to the top of the sensor.
167    pub serial_number: u64,
168    /// Indicates the shot limiting status of the sensor.
169    pub shot_limiting: ShotLimiting,
170    /// Indicates whether thermal shutdown is imminent.
171    pub shutdown_status: ShutdownStatus,
172    /// Alert flags is a bitmask of various sensor alerts.
173    pub alert_status: AlertStatus,
174}

176impl Header {
177    /// Length of the header in bytes/octets.
178    pub const LEN: usize = 32;
179}

181#[derive(Copy, Clone, Debug, Eq, PartialEq)]
182pub struct HeaderSlice<'a> {
183    slice: &'a [u8],
184}
185
186impl<'a> HeaderSlice<'a> {
187    pub fn from_slice(slice: &'a [u8]) -> Result<HeaderSlice<'a>, Error> {
188        if slice.len() < Header::LEN {
189            return Err(Error::UnexpectedEnd(slice.len()));
190        }
191
192        let packet_type = u16::from_le_bytes([slice[0], slice[1]]);
193        if packet_type != 1 {
194            return Err(Error::UnknownPacketType(packet_type));
195        }
196
197        Ok(HeaderSlice { slice })
198    }
199
200    pub fn to_header(&self) -> Header {
201        Header {
202            packet_type: self.packet_type(),
203            frame_id: self.frame_id(),
204            init_id: self.init_id(),
205            serial_number: self.serial_number(),
206            shot_limiting: self.shot_limiting(),
207            shutdown_status: self.shutdown_status(),
208            alert_status: self.alert_status(),
209        }
210    }
211
212    pub fn packet_type(&self) -> u16 {
213        u16::from_le_bytes([self.slice[0], self.slice[1]])
214    }
215
216    pub fn frame_id(&self) -> u16 {
217        u16::from_le_bytes([self.slice[2], self.slice[3]])
218    }
219
220    pub fn init_id(&self) -> u32 {
221        u32::from_le_bytes([self.slice[4], self.slice[5], self.slice[6], 0])
222    }
223
224    pub fn serial_number(&self) -> u64 {
225        u64::from_le_bytes([
226            self.slice[7],
227            self.slice[8],
228            self.slice[9],
229            self.slice[10],
230            self.slice[11],
231            0,
232            0,
233            0,
234        ])
235    }
236
237    pub fn shot_limiting(&self) -> ShotLimiting {
238        match self.slice[19] >> 4 {
239            0 => ShotLimiting::Normal,
240            1 => ShotLimiting::Imminent(self.shot_limiting_countdown()),
241            2 => ShotLimiting::Limiting(3),
242            3 => ShotLimiting::Limiting(6),
243            4 => ShotLimiting::Limiting(9),
244            5 => ShotLimiting::Limiting(12),
245            6 => ShotLimiting::Limiting(16),
246            7 => ShotLimiting::Limiting(21),
247            8 => ShotLimiting::Limiting(25),
248            9 => ShotLimiting::Limiting(27),
249            val => ShotLimiting::Invalid(val),
250        }
251    }
252
253    pub fn shutdown_status(&self) -> ShutdownStatus {
254        match self.slice[18] >> 4 {
255            0 => ShutdownStatus::Normal,
256            1 => ShutdownStatus::Imminent(self.shutdown_countdown()),
257            val => ShutdownStatus::Invalid(val),
258        }
259    }
260
261    pub fn shot_limiting_countdown(&self) -> u8 {
262        self.slice[17]
263    }
264
265    pub fn shutdown_countdown(&self) -> u8 {
266        self.slice[16]
267    }
268
269    pub fn alert_status(&self) -> AlertStatus {
270        AlertStatus::from_flags(self.slice[12])
271    }
272
273    /// Returns the column at the given index.
274    pub fn column(&self, rows: usize, col: usize) -> Result<ColumnHeaderSlice<'a>, Error> {
275        let offset = Header::LEN + (ColumnHeader::LEN + DataBlock::LEN * rows) * col;
276        ColumnHeaderSlice::from_slice(&self.slice[offset..])
277    }
278}

297#[derive(Copy, Clone, Debug, Eq, PartialEq)]
298pub struct ColumnHeaderSlice<'a> {
299    slice: &'a [u8],
300}
301
302impl<'a> ColumnHeaderSlice<'a> {
303    pub fn from_slice(slice: &'a [u8]) -> Result<ColumnHeaderSlice<'a>, Error> {
304        if slice.len() < ColumnHeader::LEN {
305            return Err(Error::UnexpectedEnd(slice.len()));
306        }
307
308        Ok(ColumnHeaderSlice { slice })
309    }
310
311    pub fn to_column(&self) -> ColumnHeader {
312        ColumnHeader {
313            timestamp: self.timestamp(),
314            measurement_id: self.measurement_id(),
315            status: self.status(),
316        }
317    }
318
319    pub fn timestamp(&self) -> u64 {
320        u64::from_le_bytes([
321            self.slice[0],
322            self.slice[1],
323            self.slice[2],
324            self.slice[3],
325            self.slice[4],
326            self.slice[5],
327            self.slice[6],
328            self.slice[7],
329        ])
330    }
331
332    pub fn measurement_id(&self) -> u16 {
333        u16::from_le_bytes([self.slice[8], self.slice[9]])
334    }
335
336    pub fn status(&self) -> bool {
337        self.slice[10] & 1 != 0
338    }
339
340    pub fn row(&self, row: usize) -> Result<DataBlock, Error> {
341        let offset = ColumnHeader::LEN + row * DataBlock::LEN;
342
343        if self.slice.len() < offset + DataBlock::LEN {
344            let delta = offset + DataBlock::LEN - self.slice.len();
345            return Err(Error::UnexpectedEnd(delta));
346        }
347
348        Ok(DataBlock {
349            range: u16::from_le_bytes([self.slice[offset], self.slice[offset + 1] & 0x7F]),
350            reflect: self.slice[offset + 2],
351            nir: self.slice[offset + 3],
352        })
353    }
354}

373pub struct FrameReader {
374    pub rows: usize,
375    pub cols: usize,
376    columns_per_packet: usize,
377    frame_id: u16,
378    /// Depth buffer owned by FrameReader - never copied
379    depth: Array2<u16>,
380    /// Reflectivity buffer owned by FrameReader - never copied
381    reflect: Array2<u8>,
382    timestamp: u64,
383    /// True when a complete frame is ready for processing
384    frame_ready: bool,
385}
386
387impl FrameReader {
388    pub fn new(lidar_data_format: &LidarDataFormat) -> Result<FrameReader, Error> {
389        if lidar_data_format.udp_profile_lidar != "RNG15_RFL8_NIR8" {
390            return Err(Error::UnsupportedFormat(
391                lidar_data_format.udp_profile_lidar.clone(),
392            ));
393        }
394
395        let cols = lidar_data_format.columns_per_frame;
396        let rows = lidar_data_format.pixels_per_column;
397        let columns_per_packet = lidar_data_format.columns_per_packet;
398
399        Ok(FrameReader {
400            rows,
401            cols,
402            columns_per_packet,
403            frame_id: 0,
404            depth: Array2::zeros((rows, cols)),
405            reflect: Array2::zeros((rows, cols)),
406            timestamp: 0,
407            frame_ready: false,
408        })
409    }
410
411    /// Process a UDP packet, accumulating data into internal buffers.
412    ///
413    /// Returns `Ok(true)` when a frame boundary is detected (new frame_id),
414    /// meaning the previous frame's data is complete and ready for processing.
415    ///
416    /// # Zero-Copy Design
417    ///
418    /// The depth and reflect buffers are owned by FrameReader and never copied.
419    /// After this returns `true`, call `depth()` and `reflect()` to get
420    /// references to the completed frame data, then call `start_new_frame()`
421    /// to prepare for the next frame.
422    pub fn update(&mut self, slice: &[u8]) -> Result<bool, Error> {
423        let header = HeaderSlice::from_slice(slice)?;
424
425        // Detect frame boundary
426        let is_new_frame = self.frame_id != header.frame_id();
427        if is_new_frame {
428            // Previous frame is complete - mark ready and update frame_id
429            self.frame_ready = true;
430            self.frame_id = header.frame_id();
431        }
432
433        // If we just detected a boundary, don't process this packet yet
434        // The caller should first consume the completed frame
435        if self.frame_ready {
436            return Ok(true);
437        }
438
439        // Process packet columns into our buffers
440        for i in 0..self.columns_per_packet {
441            let column = header.column(self.rows, i)?;
442            if column.status() {
443                let col = column.measurement_id() as usize;
444                if col < self.cols {
445                    for row in 0..self.rows {
446                        let data = column.row(row)?;
447                        self.depth[[row, col]] = data.range;
448                        self.reflect[[row, col]] = data.reflect;
449                    }
450                }
451            }
452
453            if i == 0 {
454                self.timestamp = timestamp()?;
455            }
456        }
457
458        Ok(false)
459    }
460
461    /// Start a new frame after the previous one has been consumed.
462    ///
463    /// This clears the buffers and resets the frame_ready flag.
464    pub fn start_new_frame(&mut self) {
465        self.depth.fill(0);
466        self.reflect.fill(0);
467        self.frame_ready = false;
468    }
469
470    /// Get the timestamp of the current/completed frame.
471    pub fn timestamp(&self) -> u64 {
472        self.timestamp
473    }
474
475    /// Get the frame_id of the current/completed frame.
476    pub fn frame_id(&self) -> u16 {
477        self.frame_id
478    }
479
480    /// Get reference to the depth buffer (zero-copy access).
481    pub fn depth(&self) -> &Array2<u16> {
482        &self.depth
483    }
484
485    /// Get reference to the reflect buffer (zero-copy access).
486    pub fn reflect(&self) -> &Array2<u8> {
487        &self.reflect
488    }
489}

491/// Point cloud coordinate calculator with pre-computed lookup tables.
492///
493/// Uses pre-computed coefficients in linear output order for efficient
494/// SIMD processing. All lookup tables are computed once at construction.
495pub struct FrameBuilder {
496    pub rows: usize,
497    pub cols: usize,
498    pub n_points: usize,
499    pub crop: (usize, usize),
500    /// Scratch buffer for range values (internal units before conversion)
501    pub range: Vec<f32>,
502    /// Range delta for coordinate calculation
503    range_delta: f32,
504
505    // Pre-computed coefficients in linear output order for direct indexing.
506    /// Source indices: linear index into packet depth/reflect arrays
507    src_indices: Vec<usize>,
508    /// Pre-flattened x_range coefficients (for r * x_coeff calculation)
509    x_coeff: Vec<f32>,
510    /// Pre-flattened y_range coefficients (for r * y_coeff calculation)
511    y_coeff: Vec<f32>,
512    /// Pre-flattened x_delta offsets
513    x_offset: Vec<f32>,
514    /// Pre-flattened y_delta offsets
515    y_offset: Vec<f32>,
516    /// Pre-flattened altitude coefficients (sin of altitude angle)
517    z_coeff: Vec<f32>,
518    /// beam_to_lidar Z offset (cached for SIMD broadcast)
519    beam_z: f32,
520}
521
522impl FrameBuilder {
523    pub fn new(params: &Parameters) -> Self {
524        let rows = params.lidar_data_format.pixels_per_column;
525        let cols = params.lidar_data_format.columns_per_frame;
526        let pixel_shift_by_row = params.lidar_data_format.pixel_shift_by_row.clone();
527        let column_window = params.lidar_data_format.column_window;
528        let beam_to_lidar = Array2::from_shape_vec(
529            (4, 4),
530            params.beam_intrinsics.beam_to_lidar_transform.clone(),
531        )
532        .unwrap();
533        let range_delta = (beam_to_lidar[[0, 3]].powi(2) + beam_to_lidar[[2, 3]].powi(2)).sqrt();
534
535        // The left and right columns are incomplete within the pixel shift
536        // region.  We crop out this region to return the clean subset.
537        let crop = {
538            let mut pixel_shift = pixel_shift_by_row.clone();
539            pixel_shift.sort_unstable();
540            let left = pixel_shift[pixel_shift.len() - 1];
541            let right = pixel_shift[0].abs();
542            let right = (column_window[1] - column_window[0]) as i16 - right;
543            (left as usize, right as usize)
544        };
545
546        let enc = (0..cols)
547            .map(|col| 2.0 * PI * (1.0 - col as f32 / cols as f32))
548            .collect::<Vec<_>>();
549
550        let azimuth_angles: Vec<_> = params
551            .beam_intrinsics
552            .beam_azimuth_angles
553            .iter()
554            .map(|x| -2.0 * PI * (x / 360.0))
555            .collect();
556
557        let altitude_angles: Vec<_> = params
558            .beam_intrinsics
559            .beam_altitude_angles
560            .iter()
561            .map(|x| 2.0 * PI * (x / 360.0))
562            .collect();
563
564        let mut x_range = Array2::zeros((rows, cols));
565        let mut y_range = Array2::zeros((rows, cols));
566
567        for row in 0..rows {
568            for col in 0..cols {
569                x_range[[row, col]] =
570                    (enc[col] + azimuth_angles[row]).cos() * altitude_angles[row].cos();
571                y_range[[row, col]] =
572                    (enc[col] + azimuth_angles[row]).sin() * altitude_angles[row].cos();
573            }
574        }
575
576        let x_delta: Vec<f32> = enc
577            .iter()
578            .map(|x| beam_to_lidar[[0, 3]] * x.cos())
579            .collect();
580        let y_delta: Vec<f32> = enc
581            .iter()
582            .map(|x| beam_to_lidar[[0, 3]] * x.sin())
583            .collect();
584
585        // Pre-compute altitude sin values
586        let altitude_sin: Vec<f32> = altitude_angles.iter().map(|x| x.sin()).collect();
587
588        // Pre-compute coefficients in linear output order.
589        let output_cols = crop.1 - crop.0;
590        let n_output_points = rows * output_cols;
591        let mut src_indices = Vec::with_capacity(n_output_points);
592        let mut x_coeff = Vec::with_capacity(n_output_points);
593        let mut y_coeff = Vec::with_capacity(n_output_points);
594        let mut x_offset = Vec::with_capacity(n_output_points);
595        let mut y_offset = Vec::with_capacity(n_output_points);
596        let mut z_coeff = Vec::with_capacity(n_output_points);
597
598        let col_base = column_window[0] as i16;
599        let packet_cols = column_window[1] - column_window[0];
600
601        for row in 0..rows {
602            let col_offset = (col_base - pixel_shift_by_row[row]).max(0) as usize;
603
604            for col in crop.0..crop.1 {
605                // Source index: reverse pixel shift to find packet data location
606                let shift = pixel_shift_by_row[row];
607                let src_col = ((col as i16 - shift).clamp(0, packet_cols as i16 - 1)) as usize
608                    + column_window[0];
609                src_indices.push(row * cols + src_col);
610
611                // Lookup table index with column offset adjustment
612                let lookup_col = col + col_offset;
613                x_coeff.push(x_range[[row, lookup_col]]);
614                y_coeff.push(y_range[[row, lookup_col]]);
615                x_offset.push(x_delta[lookup_col]);
616                y_offset.push(y_delta[lookup_col]);
617                z_coeff.push(altitude_sin[row]);
618            }
619        }
620
621        let beam_z = beam_to_lidar[[2, 3]];
622
623        FrameBuilder {
624            rows,
625            cols,
626            n_points: 0,
627            crop,
628            range: vec![0f32; n_output_points],
629            range_delta,
630            src_indices,
631            x_coeff,
632            y_coeff,
633            x_offset,
634            y_offset,
635            z_coeff,
636            beam_z,
637        }
638    }
639
640    /// Process depth and reflectivity data into XYZ point cloud.
641    ///
642    /// Reads directly from packet arrays, applies range scaling, computes
643    /// Cartesian coordinates using pre-computed lookup tables, and writes
644    /// directly to the output frame buffers (true zero-copy).
645    #[instrument(skip_all, fields(rows = self.rows, cols = self.cols))]
646    pub fn update_fused<F: LidarFrameWriter>(
647        &mut self,
648        depth: &Array2<u16>,
649        reflect: &Array2<u8>,
650        frame: &mut F,
651    ) {
652        let n_points = self.src_indices.len();
653        let depth_raw = depth.as_slice().expect("depth array not contiguous");
654        let reflect_raw = reflect.as_slice().expect("reflect array not contiguous");
655
656        // Get all mutable slices at once (avoids borrow checker issues)
657        let (frame_x, frame_y, frame_z, frame_intensity, frame_range) = frame.buffers_mut();
658
659        // Convert depth to range and extract intensity directly into frame
660        // Using indexed loop for clarity with multiple source arrays
661        #[allow(clippy::needless_range_loop)]
662        for i in 0..n_points {
663            let src_idx = self.src_indices[i];
664            let d = depth_raw[src_idx];
665            // Range calculation: 0 stays 0, otherwise apply scale and offset
666            self.range[i] = if d == 0 {
667                0.0
668            } else {
669                d as f32 * 8.0 - self.range_delta
670            };
671            frame_intensity[i] = reflect_raw[src_idx];
672        }
673
674        // Calculate XYZ coordinates directly into frame buffers (zero-copy)
675        self.calculate_points_fused_into(n_points, frame_x, frame_y, frame_z);
676        self.n_points = n_points;
677
678        // Copy range to frame (convert to meters)
679        for (i, frame_r) in frame_range.iter_mut().enumerate().take(n_points) {
680            *frame_r = self.range[i] * 0.001;
681        }
682
683        // Set the valid length
684        frame.set_len(n_points);
685    }
686
687    /// Calculate XYZ coordinates using NEON SIMD, writing directly to provided
688    /// output slices (true zero-copy).
689    #[cfg(target_arch = "aarch64")]
690    #[inline(never)]
691    fn calculate_points_fused_into(
692        &self,
693        len: usize,
694        out_x: &mut [f32],
695        out_y: &mut [f32],
696        out_z: &mut [f32],
697    ) {
698        const LANES: usize = 4;
699        let n_simd = len - len % LANES;
700
701        // SAFETY: NEON intrinsics are always available on aarch64.
702        unsafe {
703            let beam_z = vdupq_n_f32(self.beam_z);
704            let scale = vdupq_n_f32(0.001);
705
706            let range_ptr = self.range.as_ptr();
707            let x_coeff_ptr = self.x_coeff.as_ptr();
708            let x_offset_ptr = self.x_offset.as_ptr();
709            let y_coeff_ptr = self.y_coeff.as_ptr();
710            let y_offset_ptr = self.y_offset.as_ptr();
711            let z_coeff_ptr = self.z_coeff.as_ptr();
712            let x_out_ptr = out_x.as_mut_ptr();
713            let y_out_ptr = out_y.as_mut_ptr();
714            let z_out_ptr = out_z.as_mut_ptr();
715
716            for i in (0..n_simd).step_by(LANES) {
717                let r = vld1q_f32(range_ptr.add(i));
718
719                // X = (r * x_coeff + x_offset) * scale
720                let xc = vld1q_f32(x_coeff_ptr.add(i));
721                let xo = vld1q_f32(x_offset_ptr.add(i));
722                let x = vmulq_f32(vfmaq_f32(xo, r, xc), scale);
723                vst1q_f32(x_out_ptr.add(i), x);
724
725                // Y = (r * y_coeff + y_offset) * scale
726                let yc = vld1q_f32(y_coeff_ptr.add(i));
727                let yo = vld1q_f32(y_offset_ptr.add(i));
728                let y = vmulq_f32(vfmaq_f32(yo, r, yc), scale);
729                vst1q_f32(y_out_ptr.add(i), y);
730
731                // Z = (r * z_coeff + beam_z) * scale
732                let zc = vld1q_f32(z_coeff_ptr.add(i));
733                let z = vmulq_f32(vfmaq_f32(beam_z, r, zc), scale);
734                vst1q_f32(z_out_ptr.add(i), z);
735            }
736        }
737
738        // Scalar remainder
739        for i in n_simd..len {
740            let r = self.range[i];
741            out_x[i] = (r * self.x_coeff[i] + self.x_offset[i]) * 0.001;
742            out_y[i] = (r * self.y_coeff[i] + self.y_offset[i]) * 0.001;
743            out_z[i] = (r * self.z_coeff[i] + self.beam_z) * 0.001;
744        }
745    }
746
747    /// Calculate XYZ coordinates using portable_simd, writing directly to
748    /// provided output slices (true zero-copy).
749    #[cfg(all(feature = "portable_simd", not(target_arch = "aarch64")))]
750    #[inline(never)]
751    fn calculate_points_fused_into(
752        &self,
753        len: usize,
754        out_x: &mut [f32],
755        out_y: &mut [f32],
756        out_z: &mut [f32],
757    ) {
758        const LANES: usize = 4;
759        let n_simd = len - len % LANES;
760
761        let beam_z = Simd::<f32, LANES>::splat(self.beam_z);
762        let scale = Simd::<f32, LANES>::splat(0.001);
763
764        for i in (0..n_simd).step_by(LANES) {
765            let r = Simd::<f32, LANES>::from_slice(&self.range[i..]);
766
767            let xc = Simd::<f32, LANES>::from_slice(&self.x_coeff[i..]);
768            let xo = Simd::<f32, LANES>::from_slice(&self.x_offset[i..]);
769            let x = (r * xc + xo) * scale;
770            x.copy_to_slice(&mut out_x[i..]);
771
772            let yc = Simd::<f32, LANES>::from_slice(&self.y_coeff[i..]);
773            let yo = Simd::<f32, LANES>::from_slice(&self.y_offset[i..]);
774            let y = (r * yc + yo) * scale;
775            y.copy_to_slice(&mut out_y[i..]);
776
777            let zc = Simd::<f32, LANES>::from_slice(&self.z_coeff[i..]);
778            let z = (r * zc + beam_z) * scale;
779            z.copy_to_slice(&mut out_z[i..]);
780        }
781
782        // Scalar remainder
783        for i in n_simd..len {
784            let r = self.range[i];
785            out_x[i] = (r * self.x_coeff[i] + self.x_offset[i]) * 0.001;
786            out_y[i] = (r * self.y_coeff[i] + self.y_offset[i]) * 0.001;
787            out_z[i] = (r * self.z_coeff[i] + self.beam_z) * 0.001;
788        }
789    }
790
791    /// Calculate XYZ coordinates using scalar code, writing directly to
792    /// provided output slices (true zero-copy).
793    #[cfg(all(not(feature = "portable_simd"), not(target_arch = "aarch64")))]
794    #[inline(never)]
795    fn calculate_points_fused_into(
796        &self,
797        len: usize,
798        out_x: &mut [f32],
799        out_y: &mut [f32],
800        out_z: &mut [f32],
801    ) {
802        for i in 0..len {
803            let r = self.range[i];
804            out_x[i] = (r * self.x_coeff[i] + self.x_offset[i]) * 0.001;
805            out_y[i] = (r * self.y_coeff[i] + self.y_offset[i]) * 0.001;
806            out_z[i] = (r * self.z_coeff[i] + self.beam_z) * 0.001;
807        }
808    }
809}

941/// Maximum UDP packet size for Ouster sensors (OS1-128 at 1024 mode)
942const MAX_PACKET_SIZE: usize = 16 * 1024;
943
944/// Ouster LiDAR driver implementing the common LidarDriver trait
945///
946/// This driver wraps the FrameReader and FrameBuilder to provide
947/// a unified interface for processing Ouster UDP packets.
948///
949/// # Zero-Copy Architecture
950///
951/// The driver uses a true zero-copy design:
952/// - FrameReader owns internal depth/reflect buffers (never copied)
953/// - FrameBuilder writes directly to client-provided frame buffers
954/// - No intermediate copies or allocations during processing
955/// - Pre-allocated packet buffer handles frame boundary packets
956pub struct OusterDriver {
957    /// Frame reader for packet parsing (owns depth/reflect buffers)
958    frame_reader: FrameReader,
959    /// Frame builder for point cloud computation
960    frame_builder: FrameBuilder,
961    /// Pre-allocated buffer for packet that triggers frame boundary
962    pending_packet_buf: Vec<u8>,
963    /// Length of valid data in pending_packet_buf (0 = no pending packet)
964    pending_packet_len: usize,
965}
966
967impl OusterDriver {
968    /// Create a new Ouster driver from sensor parameters
969    pub fn new(params: &Parameters) -> Result<Self, Error> {
970        let frame_reader = FrameReader::new(&params.lidar_data_format)?;
971        let frame_builder = FrameBuilder::new(params);
972
973        Ok(Self {
974            frame_reader,
975            frame_builder,
976            pending_packet_buf: vec![0u8; MAX_PACKET_SIZE],
977            pending_packet_len: 0,
978        })
979    }
980
981    /// Get the number of rows in the frame
982    pub fn rows(&self) -> usize {
983        self.frame_builder.rows
984    }
985
986    /// Get the number of columns in the frame
987    pub fn cols(&self) -> usize {
988        self.frame_builder.cols
989    }
990
991    /// Get the crop boundaries
992    pub fn crop(&self) -> (usize, usize) {
993        self.frame_builder.crop
994    }
995}
996
997impl LidarDriver for OusterDriver {
998    fn process<F: LidarFrameWriter>(&mut self, frame: &mut F, data: &[u8]) -> Result<bool, Error> {
999        // Process pending packet from previous frame boundary first
1000        if self.pending_packet_len > 0 {
1001            let pending = &self.pending_packet_buf[..self.pending_packet_len];
1002            self.frame_reader.update(pending)?;
1003            self.pending_packet_len = 0;
1004        }
1005
1006        // Process current packet
1007        let frame_ready = self.frame_reader.update(data)?;
1008
1009        if frame_ready {
1010            // Store this packet in pre-allocated buffer (no allocation)
1011            let len = data.len().min(MAX_PACKET_SIZE);
1012            self.pending_packet_buf[..len].copy_from_slice(&data[..len]);
1013            self.pending_packet_len = len;
1014
1015            // Get zero-copy references to FrameReader's buffers
1016            let timestamp = self.frame_reader.timestamp();
1017            let frame_id = self.frame_reader.frame_id();
1018            let depth = self.frame_reader.depth();
1019            let reflect = self.frame_reader.reflect();
1020
1021            // Reset frame for new data
1022            frame.reset();
1023            frame.set_timestamp(timestamp);
1024            frame.set_frame_id(frame_id as u32);
1025
1026            // Build point cloud directly into frame buffers (true zero-copy)
1027            self.frame_builder.update_fused(depth, reflect, frame);
1028
1029            // Prepare FrameReader for next frame
1030            self.frame_reader.start_new_frame();
1031
1032            Ok(true)
1033        } else {
1034            Ok(false)
1035        }
1036    }
1037}
